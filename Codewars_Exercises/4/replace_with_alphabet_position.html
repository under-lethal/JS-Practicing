...Replace every letter with its POSITION in the ALPHABET.
If anything in (in the text) isn't a LETTER - ignore it - and don't return it.

	< div id='an'>< /div>
	< script type="text/javascript">
		let alphabet = "abcdefghijklmnopqrstuvwxyz";

		const alphabetPosition = (text) => 
		text
		.toLowerCase()
		.replace(/[^a-zA-Z]/g, "")
		.split("")
		.map((letter) => alphabet.indexOf(letter))
		.join(' ');
		document.getElementById('an').innerHTML = alphabetPosition("The sunset sets at twelve o' cloock.");
	< /script>
