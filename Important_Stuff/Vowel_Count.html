2.for of...

< div id="vc1">< /div>
< script type="text/javascript">
	let vowelCount = 0;
	const getCount1 = (str) => {
		for (const letter of str){
			if(letter.match(/[aeiou]/g)){
				vowelCount ++;
			}
		}
		return vowelCount;
	}
	document.getElementById('vc1').innerHTML = getCount1('abracadabra');

Or using reduce();

< div id="vc2">< /div>
< script type="text/javascript">	
	const getCount2 = (str) => {		
		const vowelCount2 = str.split('').reduce((acc, letter)=>{
			if(letter.match(/[aeiou]/g)){
				return acc + 1;
			}
			return acc;
		} ,0);
		return vowelCount2;
	};
	document.getElementById('vc2').innerHTML = getCount2('abracadabra');	
< /script>
